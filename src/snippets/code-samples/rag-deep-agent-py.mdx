        )

        "name": "chunk-analyst",
        "description": (

    }

    model = init_chat_model(model="google_genai:gemini-3.5-flash", temperature=0.0)

    agent = create_deep_agent(
        model=model,
        tools=[search_documentation],
        system_prompt=INSTRUCTIONS,
        subagents=[chunk_analyst_subagent],
    )
