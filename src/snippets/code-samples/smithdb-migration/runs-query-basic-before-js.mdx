```ts
import { Client } from "langsmith";

const client = new Client();
const runs = client.listRuns({
  projectName: "default",
  runType: "llm",
  filter: 'and(eq(status, "success"), gt(total_tokens, 100))',
  limit: 100,
});
for await (const run of runs) {
  console.log(run.id, run.name, run.status);
}
```
