```go
import (
	"context"
	"fmt"

	"github.com/langchain-ai/langsmith-go"
)

ctx := context.Background()
client := langsmith.NewClient()

response, err := client.Runs.Query(ctx, langsmith.RunQueryParams{
	ID: langsmith.F([]string{"<run-id-1>", "<run-id-2>"}),
})
if err != nil {
	panic(err.Error())
}
for _, run := range response.Runs {
	fmt.Println(run.ID, run.Name, run.Status)
}
```
