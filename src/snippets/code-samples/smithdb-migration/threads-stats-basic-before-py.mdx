from langsmith import Client

client = Client()
project = client.read_project(project_name="default")
thread_id = "<thread-id>"

    project_ids=[project.id],
    filter=f'eq(thread_id, "{thread_id}")',
    is_root=True,
)
print(stats["run_count"], stats["total_tokens"], stats["total_cost"])
