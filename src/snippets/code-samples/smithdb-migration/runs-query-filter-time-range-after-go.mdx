```go After
import (
	"context"
	"fmt"
	"time"

	"github.com/langchain-ai/langsmith-go"
)

ctx := context.Background()
client := langsmith.NewClient()

pager := client.Runs.QueryV2AutoPaging(ctx, langsmith.RunQueryV2Params{
	ProjectIDs:   langsmith.F([]string{"<project-id>"}),
	MinStartTime: langsmith.F(time.Now().Add(-24 * time.Hour)),
	RunType:      langsmith.F(langsmith.RunQueryV2ParamsRunTypeLLM),
})
for pager.Next() {
	run := pager.Current()
	fmt.Println(run.ID, run.Name, run.Status)
}
if err := pager.Err(); err != nil {
	panic(err.Error())
}
```
