```go
import (
	"context"
	"fmt"

	"github.com/langchain-ai/langsmith-go"
)

ctx := context.Background()
client := langsmith.NewClient()

filterStr := "and(eq(metadata_key, 'user_id'), eq(metadata_value, 'u_123'))"
response, err := client.Runs.Query(ctx, langsmith.RunQueryParams{
	Session: langsmith.F([]string{"<project-id>"}),
	Filter:  langsmith.F(filterStr),
})
if err != nil {
	panic(err.Error())
}
for _, run := range response.Runs {
	fmt.Println(run.ID, run.Name, run.Status)
}
```
